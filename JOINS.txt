
JOINS IN SQL
--------------------
SELECT * FROM HR.EMPLOYEES;

SELECT * FROM HR.DEPARTMENTS;

SELECT * FROM HR.LOCATIONS;

-- 4TH PPT (JOINS)
-- CARTESIAN PRODUCT

SELECT * 
FROM HR.EMPLOYEES 
CROSS JOIN HR.DEPARTMENTS;


--EQUI JOIN (INNER JOIN)

SELECT *
FROM HR.EMPLOYEES INNER JOIN HR.DEPARTMENTS
ON HR.EMPLOYEES.MANAGER_ID = HR.DEPARTMENTS.MANAGER_ID;

SELECT *
FROM HR.EMPLOYEES INNER JOIN HR.DEPARTMENTS
ON HR.EMPLOYEES.DEPARTMENT_ID = HR.DEPARTMENTS.DEPARTMENT_ID;


SELECT E.EMPLOYEE_ID, E.MANAGER_ID, E.DEPARTMENT_ID, D.DEPARTMENT_NAME
FROM HR.EMPLOYEES E
INNER JOIN HR.DEPARTMENTS D ON E.DEPARTMENT_ID = D.DEPARTMENT_ID
WHERE D.DEPARTMENT_ID  = 30;


SELECT E.EMPLOYEE_ID , E.MANAGER_ID,E.DEPARTMENT_ID,
D.DEPARTMENT_ID, D.DEPARTMENT_NAME 
FROM HR.EMPLOYEES E  INNER JOIN HR.DEPARTMENTS D
ON E.DEPARTMENT_ID = D.DEPARTMENT_ID
WHERE D.DEPARTMENT_ID  = 30;


-- OUTER  JOIN
-- LEFT OUTER JOIN
/* IT WILL GIVE THE RESULT OF ENTIRE LEFT SIDE TABLE ALONG WITH THE MATCHING RECORDS OF RIGHT SIDE TABLE*/


SELECT E.EMPLOYEE_ID,E.FIRST_NAME,E.LAST_NAME,E.SALARY,E.MANAGER_ID,D.MANAGER_ID
FROM HR.EMPLOYEES E LEFT JOIN HR.DEPARTMENTS D
ON E.MANAGER_ID = D.MANAGER_ID
WHERE D.MANAGER_ID IN (103,203,216);

-- RIGHT OUTER JOIN
/* IT WILL GIVE THE RESULT AS ENTRE RIGHT SIDE TABLE ALONG WITH THE MATCHING RECORDS OF LEFT SIDE TABLE*/

SELECT E.EMPLOYEE_ID,E.FIRST_NAME,E.LAST_NAME,E.MANAGER_ID,D.DEPARTMENT_NAME,D.DEPARTMENT_ID,
D.MANAGER_ID,D.LOCATION_ID
FROM HR.EMPLOYEES E RIGHT JOIN HR.DEPARTMENTS D
ON E.DEPARTMENT_ID = D.DEPARTMENT_ID
WHERE E.DEPARTMENT_ID = 20;

SELECT E.EMPLOYEE_ID,E.FIRST_NAME,E.LAST_NAME,E.MANAGER_ID,D.DEPARTMENT_NAME,D.DEPARTMENT_ID,
D.MANAGER_ID,D.LOCATION_ID
FROM HR.EMPLOYEES E RIGHT JOIN HR.DEPARTMENTS D
ON E.DEPARTMENT_ID = D.DEPARTMENT_ID
WHERE D.DEPARTMENT_NAME = 'Administration'  AND D.DEPARTMENT_ID = 10;

-- FULL OUTER JOIN
/* FULL OUTER JOIN = LEFT OUTER + RIGHT OUTER */

SELECT E.EMPLOYEE_ID,E.FIRST_NAME,E.LAST_NAME,E.MANAGER_ID,D.DEPARTMENT_NAME,D.DEPARTMENT_ID,
D.MANAGER_ID,D.LOCATION_ID
FROM HR.EMPLOYEES E FULL OUTER JOIN HR.DEPARTMENTS D
ON E.DEPARTMENT_ID = D.DEPARTMENT_ID
WHERE E.DEPARTMENT_ID = 20;
